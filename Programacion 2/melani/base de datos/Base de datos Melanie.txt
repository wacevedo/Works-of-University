CREATE TABLE ESTUDIANTE(
ID_ESTUDIANTE NUMBER,
NOMBRE VARCHAR2(50),
APELLIDO VARCHAR2(50),
MATRICULA VARCHAR2(9),
ID_PROYECTO NUMBER);

CREATE TABLE LENGUAJE(
ID_LENGUAJE NUMBER,
DESCRIPCION VARCHAR2(200));

CREATE TABLE ETAPA(
ID_ETAPA NUMBER,
NOMBRE VARCHAR2(50),
DESCRIPCION VARCHAR2(200),
ESTADO VARCHAR2(50),
FECHA_INICIO DATE,
FECHA_FINAL DATE,
ID_PROYECTO NUMBER);

CREATE TABLE EMPRESA(
ID_EMPRESA NUMBER,
NOMBRE VARCHAR2(50),
DESCRIPCION VARCHAR2(200));

CREATE TABLE PROFESOR(
ID_PROFESOR NUMBER,
NOMBRE VARCHAR2(50),
APELLIDO VARCHAR2(50),
CEDULA VARCHAR2(50),
ESTADO VARCHAR2(50));

CREATE TABLE PROYECTO(
ID_PROYECTO NUMBER,
NOMBRE VARCHAR2(50),
TIPO_PROYECTO VARCHAR2(60),
ESTADO VARCHAR2 (50),
CANTIDAD_ESTUD NUMBER,
FECHA_ENTREGA DATE,
ID_EMPRESA NUMBER,
ID_PROFESOR NUMBER);

CREATE TABLE PROYECTO_LENGUAJE(
ID_PL NUMBER,
ID_PROYECTO NUMBER,
ID_LENGUAJE NUMBER);
