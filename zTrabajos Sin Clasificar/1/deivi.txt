CREATE TABLE PROFESOR(ID_PROFESOR NUMBER NOT NULL,NOMBRE varchar2(50)NOT NULL,
DIRECCION varchar2(200) NOT NULL,TELEFONO varchar2(12),GENERO CHAR(1)NOT NULL,
CONSTRAINT PK_ID_PROFESOR primary key(ID_PROFESOR));
 
CREATE TABLE CURSOS(ID_CURSO NUMBER NOT NULL, DESCRIPCION varchar2(50) NOT NULL,HORAS NUMBER NOT NULL, ID_PROFESOR NOT NULL,
CONSTRAINT PK_ID_CURSOS primary key(ID_CURSO),
CONSTRAINT FK_IDPROFESOR FOREIGN KEY(ID_PROFESOR) REFERENCES PROFESOR(ID_PROFESOR));

CREATE TABLE ALUMNO(ID_ALUMNO NUMBER NOT NULL, NOMBRE varchar2(50)NOT NULL, CEDULA varchar2(11)NOT NULL,
FECHANAC DATE, ID_CURSO NUMBER NOT NULL,
CONSTRAINT PK_ID_ALUMNO primary key(ID_ALUMNO),
CONSTRAINT FK_CURSO foreign key(ID_CURSO) REFERENCES CURSOS(ID_CURSO));

CREATE TABLE ASIGNATURA(ID_ASIGNATURA varchar2(3) NOT NULL, DESCRIPCION varchar2(50)NOT NULL, CREDITOS NUMBER NOT NULL,
CONSTRAINT PK_ID_ASIGNATURA primary key(ID_ASIGNATURA));

CREATE TABLE CURSO_ASIGNATURA(ID_ASIGNATURA varchar2(3), ID_CURSO NUMBER,
CONSTRAINT PK_IDASIGNATURA primary key(ID_ASIGNATURA,ID_CURSO));
