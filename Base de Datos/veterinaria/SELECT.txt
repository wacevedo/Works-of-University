/*SELECT 1*/
select I.NOMBRE, COUNT(ma.id_mascota)
from INTEGRANTES I, CLIENTE_INTEGRANTES CI, CLIENTE CL, MASCOTA MA
WHERE CI.ID_CLIENTE=CL.ID_CLIENTE 
AND CI.ID_INTEGRANTES=I.ID_INTEGRANTES
AND MA.ID_CLIENTE=CL.ID_CLIENTE
GROUP BY I.NOMBRE;


/*SELECT 2*/
select count(*)
FROM calendario ca, mascota ma
WHERE ca.id_mascota=ma.id_mascota 
AND ca.fecha_vacuna between '01-aug-14' and '31-aug-14';

/*SELECT 3*/
SELECT CA.ESPECIE, COUNT (CA.ESPECIE) 
FROM MASCOTA CA
GROUP BY(CA.ESPECIE);

/*SELECT 4*/
SELECT CA.APELLIDO, COUNT(CU.ID_MASCOTA)
FROM CLIENTE CA, MASCOTA CU
WHERE CA.ID_CLIENTE=CU.ID_CLIENTE
GROUP BY(CA.APELLIDO)
HAVING COUNT(CU.ID_MASCOTA)>=2;

/*SELECT 5*/
SELECT CA.NOMBRE, COUNT(*)
FROM VETERINARIO CA, HISTORIAL CU 
WHERE CA.ID_VETERINARIO=CU.ID_VETERINARIO 
AND cu.fecha_enfermo between '01-JUN-14' AND '30-JUN-14'
GROUP BY(CA.NOMBRE);
